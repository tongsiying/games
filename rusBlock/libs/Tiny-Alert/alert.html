创建一个对话框，不加遮罩

            $.dialog({
                    content : '对话框内容',
                    title : 'ok',
                    ok : function() {
                        alert('我是确定按钮，回调函数返回false时不会关闭对话框。');
                        return false;
                    },
                    cancel : function() {
                        alert('我是取消按钮');
                    },
                    lock : false
                });
